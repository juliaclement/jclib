52 template<typename T> class yieldingIteratorHelper : public CountedPointerTarget {
53 public:
54 /**
55 * Called to advance the iterator to its next state
56 * @param value that will be returned by * iterator. I/O
57 * @param state used by helper to determine where it is. Setting
58 * state to -1 signals end() is true
59 */
60 virtual void next( T & value, int & state ) = 0;
61 /**
62 * Called when iterator is created to set initial state
63 * Optional: Default action is to call next() with state = 0;
64 * @param value that will be returned by * iterator. I/O
65 * @param state used by helper to determine where it is.
66 * There are two special values:<ul><li>-1: End of range encountered</li>
67 * <li>0 beginning of sequence required. Either the iterator is
68 * new or has been reset</li></ul>
69 */
70 virtual void begin( T & value, int & state ) {
71 state = 0;
72 next( value, state );
73 }
74 /**
75 * Called when state is set to -1 (end) either by the user
76 * or on return from next()
77 * @param value that will be returned by * iterator. I/O
78 */
79 virtual void ending( T & value ) {
80 }
81 /**
82 * Called by the yieldingIterator copy constructor. If
83 * this object has state stored outside the value and state
84 * parameters to next(), it should return a new object, otherwise
85 * it's fine to return ourselves
86 * @return either ourself or a clone of ourself
87 */
88 virtual yieldingIteratorHelper<T> * maybeClone() { return this; }
89 };

95 template<typename T> class yieldingIterator
96 : public std::iterator<std::forward_iterator_tag,
97 T,
98 ptrdiff_t,
99 T*,
100 T&> {
101 public:
102 enum { yiEof = -1, yiBegin = 0 };
103 protected:
104 // don't change the order of these members,
105 // helper_ must be constructed before state_
106 CountedPointer< yieldingIteratorHelper<T> > helper_;
107 int state_;
108 T value_;
109 public:
110 yieldingIterator<T> operator++(int postfix) {
111 yieldingIterator<T> answer = *this;
112 helper_->next(value_, state_);
113 return answer;
114 }
115 yieldingIterator<T> & operator++() {
116 helper_->next(value_, state_);
117 if( state_ == yiEof)
118 helper_->ending(value_);
119 return *this;
120 }
121 T & operator*() const { return const_cast<T&>(value_); }
122 T * operator->() const { return &const_cast<T&>(value_); }
123
124 /**
125 * Reset the iterator to the beginning of its sequence
126 * requires the helper to act appropriately to the begin() call
127 */
128 void set_begin() {
129 if( helper_.isset()) {
130 state_ = yiBegin;
131 helper_->begin(value_, state_);
132 } else {
133 state_ = yiEof;
134 }
135 }
136
137 /**
138 * Checks if the iterator is at the end of its range<br/>
139 * Quick test for when container.end() isn't accessible.<br/>
140 * Once end is set, helper_->next() won't be called until the
141 * iterator is reset by a call to set_begin() or an assignment.
142 * @return boolean
143 */
144 inline bool is_end() const { return state_ == yiEof; }
145
146 /**
147 * Sets the internal state to end of sequence<br/>
148 * Once end is set, helper_->next() won't be called until the
149 * iterator is reset by a call to set_begin() or an assignment.
150 */
151 void set_end() {
152 state_ = yiEof;
153 if( helper_.isset())
154 helper_->ending( value_ );
155 }
156
157 inline bool operator==( const yieldingIterator & rhs ) const {
158 return state_ == rhs.state_ &&
159 (state_ < 1 || value_ == rhs.value_);
160 }
161
162 inline bool operator!=( const yieldingIterator & rhs ) const {
163 return ! ( * this == rhs );
164 }
165
166 /** Variadic constructor relies on the helper being able to
167 * accept const & arguments in all positions. If this ever turns
168 * out to be not so we may need to subclass or even split this class
169 *
170 * @param helper
171 * @param args for the constructor of type T
172 */
173 template< typename ... Types >
174 yieldingIterator( class yieldingIteratorHelper<T> * helper,
175 const Types&...args )
176 : helper_( helper )
177 , state_( yiBegin)
178 , value_(args...)
179 {
180 if( helper_.isset() ) {
181 helper_->begin( value_, state_);
182 } else {
183 state_ = yiEof;
184 // No helper so can't call ending())
185 }
186 }
187
188 /**
189 * Constructor to satisfy the default constructable requirement
190 * for iterators.
191 */
192 yieldingIterator( )
193 : helper_( nullptr )
194 , state_( yiEof )
195 , value_( )
196 {
197 }
198
199 /**
200 * Copy constructor. NB: If the source iterator has ended, the
201 * ending() method will be called. This means it is possible for
202 * an iterator that never had a begin() to end().
203 * @param old cource of copy.
204 */
205 yieldingIterator( const yieldingIterator<T> & old )
206 : helper_( old.helper_.isnull() ? nullptr : old.helper_->maybeClone() )
207 , state_( old.helper_.isnull() ? yiEof : old.state_ )
208 , value_( old.value_)
209 {
210 if( helper_.isset() && state_ == yiEof )
211 helper_->ending(value_);
212 }
213
214 /**
215 * Assignment operator. NB: If the source iterator has ended, the
216 * ending() method will be called. This means it is possible for
217 * an iterator that never had a begin() to end().
218 * @param old cource of copy.
219 */
220 yieldingIterator<T> & operator=( const yieldingIterator<T> & old ) {
221 helper_ = old.helper_.isnull() ? nullptr : old.helper_->maybeClone();
222 state_ = old.helper_.isnull() ? yiEof : old.state_;
223 value_ = old.value_;
224 if( helper_.isset() && state_ == yiEof )
225 helper_->ending(value_);
226 return *this;
227 }
228 };

230 /**
231 * Emulates a container object providing begin & end iterators
232 * suitable for feeding to for( var : container )
233 */
234 template<typename T> class yieldingContainer {
235 public:
236 CountedPointer< yieldingIteratorHelper< T > > helper_;
237 typedef class yieldingIterator<T> iterator_T;
238 public:
239 /**
240 * instantiate and return an iterator set to the start of the range
241 * @return the iterator
242 */
243 iterator_T begin() {
244 iterator_T answer( helper_.isnull() ? nullptr : helper_->maybeClone() );
245 answer.set_begin();
246 return answer;
247 }
248 /**
249 * return an iterator set to the end of the range. Only one iterator
250 * per class is actually created
251 * @return the iterator
252 */
253 iterator_T const & end() {
254 // Being able to use a static end_ relies on the
255 // == operator considering only state_ when it is yiEof.
256 static iterator_T end_(nullptr);
257 return end_;
258 }
259 /**
260 * Constructor<br/>
261 * I have deliberately not created a default constructor.<br/>
262 * Given that instances can be constructed with a nullptr (and a
263 * test exercises that) there is no functional reason for not
264 * having one, I simply can't see a purpose for it. JC Oct 2017
265 * @param helper. If a nullptr, an assignment must be made before
266 * calling begin() or bad things will happen
267 */
268 yieldingContainer ( yieldingIteratorHelper< T > * helper )
269 : helper_( helper )
270 {
271 }
272
273
274 /**
275 * Copy constructor.
276 * @param source object
277 */
278 yieldingContainer ( const yieldingContainer< T > & old )
279 // I'm not sure if a maybeClone is required I believe not as it
280 // should happen for every iterator created through begin()
281 : helper_( old.helper_ )
282 {}
283
284 virtual ~yieldingContainer() {
285 }
286
287 /**
288 * Assignment.
289 * @param source object
290 */
291 yieldingContainer<T> & operator = ( const yieldingContainer< T > & old ) {
292 // see note on maybeClone in the copy constructor
293 helper_ = old.helper_;
294 return *this;
295 }
296 };
