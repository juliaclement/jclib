50 template<typename T> class yieldingIteratorHelper : public CountedPointerTarget {
51 public:
52 /**
53 * Called to advance the iterator to its next state
54 * @param value that will be returned by * iterator. I/O
55 * @param state used by helper to determine where it is. Setting
56 * state to -1 signals end() is true
57 */
58 virtual void next( T & value, int & state ) = 0;
59 /**
60 * Called when iterator is created to set initial state
61 * Optional: Default action is to call next() with state = 0;
62 * @param value that will be returned by * iterator. I/O
63 * @param state used by helper to determine where it is.
64 * There are two special values:<ul><li>-1: End of range encountered</li>
65 * <li>0 beginning of sequence required. Either the iterator is
66 * new or has been reset</li></ul>
67 */
68 virtual void begin( T & value, int & state ) {
69 state = 0;
70 next( value, state );
71 }
72 /**
73 * Called when state is set to -1 (end) either by the user
74 * or on return from next()
75 * @param value that will be returned by * iterator. I/O
76 */
77 virtual void ending( T & value ) {
78 }
79 /**
80 * Called by the yieldingIterator copy constructor. If
81 * this object has state stored outside the value and state
82 * parameters to next(), it should return a new object, otherwise
83 * it's fine to return ourselves
84 * @return either ourself or a clone of ourself
85 */
86 virtual yieldingIteratorHelper<T> * maybeClone() { return this; }
87 };

93 template<typename T> class yieldingIterator
94 : public std::iterator<std::forward_iterator_tag,
95 T,
96 ptrdiff_t,
97 T*,
98 T&> {
99 public:
100 enum { yiEof = -1, yiBegin = 0 };
101 protected:
102 // don't change the order of these members,
103 // helper_ must be constructed before state_
104 CountedPointer< yieldingIteratorHelper<T> > helper_;
105 int state_;
106 T value_;
107 public:
108 yieldingIterator<T> operator++(int postfix) {
109 yieldingIterator<T> answer = *this;
110 helper_->next(value_, state_);
111 return answer;
112 }
113 yieldingIterator<T> & operator++() {
114 helper_->next(value_, state_);
115 if( state_ == yiEof)
116 helper_->ending(value_);
117 return *this;
118 }
119 T & operator*() const { return const_cast<T&>(value_); }
120 T * operator->() const { return &const_cast<T&>(value_); }
121
122 /**
123 * Reset the iterator to the beginning of its sequence
124 * requires the helper to act appropriately to the begin() call
125 */
126 void set_begin() {
127 if( helper_.isset()) {
128 state_ = yiBegin;
129 helper_->begin(value_, state_);
130 } else {
131 state_ = yiEof;
132 }
133 }
134
135 /**
136 * Checks if the iterator is at the end of its range<br/>
137 * Quick test for when container.end() isn't accessible.<br/>
138 * Once end is set, helper_->next() won't be called until the
139 * iterator is reset by a call to set_begin() or an assignment.
140 * @return boolean
141 */
142 inline bool is_end() const { return state_ == yiEof; }
143
144 /**
145 * Sets the internal state to end of sequence<br/>
146 * Once end is set, helper_->next() won't be called until the
147 * iterator is reset by a call to set_begin() or an assignment.
148 */
149 void set_end() {
150 state_ = yiEof;
151 if( helper_.isset())
152 helper_->ending( value_ );
153 }
154
155 inline bool operator==( const yieldingIterator & rhs ) const {
156 return state_ == rhs.state_ &&
157 (state_ < 1 || value_ == rhs.value_);
158 }
159
160 inline bool operator!=( const yieldingIterator & rhs ) const {
161 return ! ( * this == rhs );
162 }
163
164 /** Variadic constructor relies on the helper being able to
165 * accept const & arguments in all positions. If this ever turns
166 * out to be not so we may need to subclass or even split this class
167 *
168 * @param helper
169 * @param args for the constructor of type T
170 */
171 template< typename ... Types >
172 yieldingIterator( class yieldingIteratorHelper<T> * helper,
173 const Types&...args )
174 : helper_( helper )
175 , state_( yiBegin)
176 , value_(args...)
177 {
178 if( helper_.isset() ) {
179 helper_->begin( value_, state_);
180 } else {
181 state_ = yiEof;
182 // No helper so can't call ending())
183 }
184 }
185
186 /**
187 * Constructor to satisfy the default constructable requirement
188 * for iterators.
189 */
190 yieldingIterator( )
191 : helper_( nullptr )
192 , state_( yiEof )
193 , value_( )
194 {
195 }
196
197 /**
198 * Copy constructor. NB: If the source iterator has ended, the
199 * ending() method will be called. This means it is possible for
200 * an iterator that never had a begin() to end().
201 * @param old cource of copy.
202 */
203 yieldingIterator( const yieldingIterator<T> & old )
204 : helper_( old.helper_.isnull() ? nullptr : old.helper_->maybeClone() )
205 , state_( old.helper_.isnull() ? yiEof : old.state_ )
206 , value_( old.value_)
207 {
208 if( helper_.isset() && state_ == yiEof )
209 helper_->ending(value_);
210 }
211
212 /**
213 * Assignment operator. NB: If the source iterator has ended, the
214 * ending() method will be called. This means it is possible for
215 * an iterator that never had a begin() to end().
216 * @param old cource of copy.
217 */
218 yieldingIterator<T> & operator=( const yieldingIterator<T> & old ) {
219 helper_ = old.helper_.isnull() ? nullptr : old.helper_->maybeClone();
220 state_ = old.helper_.isnull() ? yiEof : old.state_;
221 value_ = old.value_;
222 if( helper_.isset() && state_ == yiEof )
223 helper_->ending(value_);
224 return *this;
225 }
226 };

228 /**
229 * Emulates a container object providing begin & end iterators
230 * suitable for feeding to for( var : container )
231 */
232 template<typename T> class yieldingContainer {
233 public:
234 CountedPointer< yieldingIteratorHelper< T > > helper_;
235 typedef class yieldingIterator<T> iterator_T;
236 public:
237 /**
238 * instantiate and return an iterator set to the start of the range
239 * @return the iterator
240 */
241 iterator_T begin() {
242 iterator_T answer( helper_.isnull() ? nullptr : helper_->maybeClone() );
243 answer.set_begin();
244 return answer;
245 }
246 /**
247 * return an iterator set to the end of the range. Only one iterator
248 * per class is actually created
249 * @return the iterator
250 */
251 iterator_T const & end() {
252 // Being able to use a static end_ relies on the
253 // == operator considering only state_ when it is yiEof.
254 static iterator_T end_(nullptr);
255 return end_;
256 }
257 /**
258 * Constructor<br/>
259 * I have deliberately not created a default constructor.<br/>
260 * Given that instances can be constructed with a nullptr (and a
261 * test exercises that) there is no functional reason for not
262 * having one, I simply can't see a purpose for it. JC Oct 2017
263 * @param helper. If a nullptr, an assignment must be made before
264 * calling begin() or bad things will happen
265 */
266 yieldingContainer ( yieldingIteratorHelper< T > * helper )
267 : helper_( helper )
268 {
269 }
270
271
272 /**
273 * Copy constructor.
274 * @param source object
275 */
276 yieldingContainer ( const yieldingContainer< T > & old )
277 // I'm not sure if a maybeClone is required I believe not as it
278 // should happen for every iterator created through begin()
279 : helper_( old.helper_ )
280 {}
281
282 virtual ~yieldingContainer() {
283 }
284
285 /**
286 * Assignment.
287 * @param source object
288 */
289 yieldingContainer<T> & operator = ( const yieldingContainer< T > & old ) {
290 // see note on maybeClone in the copy constructor
291 helper_ = old.helper_;
292 return *this;
293 }
294 };
