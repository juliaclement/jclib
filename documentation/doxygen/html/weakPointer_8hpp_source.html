33/**
34 * Implementation of a weak pointer.
35 * jclib::WeakPointer is a smart pointer that holds a non-owning ("weak")
36 * reference to an object that is managed by jclib::CountedPointer.
37 * like std::weak_ptr, jclib::WeakPointer models temporary ownership:
38 * when an object needs to be available only if it exists, and it may be
39 * deleted at any time by other code.
40 * Like std::weak_ptr, jclib::WeakPointer is intended to break circular
41 * references in pointers..
42
43 * This is not the same as std::weak_ptr, details on that class are at
44 * http://en.cppreference.com/w/cpp/memory/weak_ptr
45 * Major differences:
46 * * std::weak_ptr is related to std::shared pointer, jclib::WeakPointer is
47 * related to jclib::CountedPointer
48 * * std::weak_pointer may hold references to any object std::shared_pointer
49 * may hold references to. jclib::WeakPointer may only hold references
50 * to objects that publicly inherit from jclib::WeakPointerTargetBase. Usually
51 * something that publicly inherits from WeakPointerTarget<T>
52 * * jclib::WeakPointer may hold a nullptr or a value returned from
53 * WeakPointerTarget<T>::getWeakPointer()
54 * * std::weak_ptr must be converted to std::shared_ptr in order to access the
55 * referenced object. jclib::WeakPointer may be directly dereferenced to a
56 * native pointer. That said, unless the pointer is very short lived,
57 * converting to a CountedPointer is usually safer.
58 * Todo: Consider if WeakPointerTargetBase should be an interface
59 */

66/**
67 * All jclib::WeakPointers are CountedPointers of this class.
68 * It is created, initialised and ultimately cleared by the target
69 */
70class WeakPointerBase {
71private:
72 int countedPointers_;
73public:
74 WeakPointerTargetBase * wtb_target_;
75
76 inline void CountedPointerAttach() { ++countedPointers_; }
77 inline void CountedPointerDetach() { if( !--countedPointers_ ) delete this;}
78 WeakPointerBase( WeakPointerTargetBase * target) : countedPointers_( 0 ), wtb_target_(target) {}
79 // copy constructor. Not required, but included as failsafe
80 WeakPointerBase( WeakPointerBase const &) : countedPointers_( 0 ) {}
81};

85template< typename T> class WeakPointer {
86 public:
87 typedef CountedPointer< class WeakPointerBase > pointer_t;
88 private:
89 pointer_t basePointer_;
90 public:
91 /**
92 * Test if the pointer is set to nullptr
93 * @return true if either the internal pointer or the WeakPointerBase is nullptr
94 **/
95 bool isnull() const {
96 return basePointer_.isnull() ||
97 basePointer_->wtb_target_ == nullptr;
98 }
99 /**
100 * Test if the pointer is set to an object
101 * * @return true if neither the internal pointer or the WeakPointerBase is nullptr
102 **/
103 bool isset() const {
104 return basePointer_.isset() &&
105 basePointer_->wtb_target_ != nullptr;
106 }
107
108 /**
109 * Obtain a native C++ pointer to the object
110 * @return the address of the object pointed at
111 */
112 T * get() const {
113 return
114 isnull()
115 ? nullptr
116 : static_cast<T*> ( const_cast<class WeakPointerTargetBase *>(basePointer_->wtb_target_) ); } //const_cast< T *>((T*)(basePointer_.get())); }
117 /**
118 * retrieve a reference to the address of the object pointed at
119 */
120 T & operator *() const {
121 return *get();//* const_cast< T *>((T*)(basePointer_->target_));
122 }
123 /**
124 * retrieve a reference to the address of the object pointed at
125 */
126 T * operator ->() const { return get(); } //return const_cast< T *>((T*)(basePointer_->target_)); }
127 /**
128 * Set to point to a different target. Attaching / Detaching as requried
129 * occasionaally useful as a replacement for the assignment operator.
130 * @param ptr the new value to set
131 */
132 inline void set( const WeakPointer<T> & old) {
133 basePointer_.set(old.basePointer_);
134 }
135 /**
136 * Set the internal value. Attaching / Detaching as requried
137 * @param ptr the new value to set
138 * @return ourself
139 */
140 class WeakPointer<T> & operator =( const WeakPointer<T> & ptr ) {
141 basePointer_ = ptr.basePointer_;
142 return *this;
143 }

185 * @param ignored as we create our own pointer_t we don't use

191 }
