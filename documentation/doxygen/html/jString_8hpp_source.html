59 struct jStringBase {
60 /**
61 * A class to store the string data.
62 */
63 private:
64 friend class jclib::jString;
65 friend class jclib::CountedPointer<jStringBase>; // = jString's parent class;
66 friend std::ostream& operator<<(std::ostream& os, const class jclib::jString & string);
67 // no copy constructor
68 jStringBase( const jStringBase & ) = delete;
69 // suppress assignment into the string
70 jStringBase& operator = (const jStringBase&) = delete;
71 // Implement the requirements for CountedPointer
72 int counter_;
73 inline void CountedPointerAttach() { ++counter_; }
74 inline void CountedPointerDetach() { if( !--counter_) delete this;}
75 // ctor private so only jString can access it
76 jStringBase( const char * data, std::size_t len )
77 : counter_(0)
78 #ifdef SAFE_JSTRING
79 , data_( new char[len+1])
80 #endif
81 , len_( len )
82 {
83 strncpy(data_, data, len);
84 data_[len] = '\0';
85 }
86 /** create a jStringBase with data */
87 static jStringBase * create(const char * data, std::size_t len) {
88 #ifdef SAFE_JSTRING
89 // Double create on heap
90 return new jStringBase(data, len);
91 #else
92 void * new_store = new unsigned char[len + sizeof(jStringBase)];
93 jStringBase * new_object= new (new_store) jStringBase(data, len);
94 return new_object;
95 #endif
96 }
97 /** create an empty jStringBase
98 * used by join() to create a place to write the answer
99 * into otherwise a bug in calling routine unless len == 0
100 * when it has the same effect as jStringBase( "", 0 )
101 */
102 static jStringBase * create(std::size_t len) {
103 #ifdef SAFE_JSTRING
104 // Double create on heap
105 return new jStringBase(data, len);
106 #else
107 void * new_store = new unsigned char[len + sizeof(jStringBase)];
108 jStringBase * new_object= new (new_store) jStringBase("", len);
109 return new_object;
110 #endif
111 }
112 #ifdef SAFE_JSTRING
113 virtual ~jStringBase() {
114 delete data_;
115 }
116 #endif
117 std::size_t len_;
118 #ifdef SAFE_JSTRING
119 // Not the most efficient, but portable.
120 char * data_;
121 #else
122 // **** This MUST be the last data member in the struct as
123 // we deliberately allocate padding then overrun the declared
124 // length of the array.
125 //
126 // I think it's safe as:
127 // C99 requires that struct members are laid out in
128 // declaration order:
129 // "Within a structure object, the non-bit-field members
130 // and the units in which bit-fields reside have addresses
131 // that increase in the order in which they are declared"
132 // C++ probably requires that in a standard-layout struct
133 // object struct members are laid out in declaration order.
134 // Problem is I can't seem to locate it in an actual C++ standard
135 // but every compiler I know works that way.
136 //
137 // If you get strange crashes, try defining SAFE_JSTRING above
138 char data_[1];
139 #endif
140 public:
141 operator const char *() const { return data_;};
142 auto len() const { return len_;}
143 };
144 };

226 // string concatenation
227 jString operator + (const jString &rhs) {
228 jStringBase * lhs_base = get();
229 if( lhs_base->len_ == 0 )
230 return rhs;
231 if( rhs.len() == 0 )
232 return *this;
233 auto len = lhs_base->len_ + rhs->len_;
234 jStringBase * new_base = jStringBase::create(lhs_base->data_,len );
235 strncpy(new_base->data_ + lhs_base->len_,rhs->data_, rhs->len_ );
236 new_base->data_[len]='\0';
237 return new_base;
238 }
239 // extract substrings (left, right, any) with
240 // silent bounds checks.
241 // E.g. if z=="123", z.substr(2,999) will quietly return "3"
242 // NB: Const but when the whole string is to be
243 // returned, we return a copy of ourself not a
244 // whole new string. This saves creating a new
245 // jStringBase.
246 jString left(std::size_t len) const {
247 auto ilen=std::min(len,get()->len_);
248 if(ilen == get()->len_)
249 return *this;
250 return jStringBase::create(get()->data_, ilen );
251 }
252 jString right(std::size_t len) const {
253 auto ilen=std::min(len,get()->len_);
254 if(ilen == get()->len_)
255 return *this;
256 auto start=get()->len_-ilen;
257 return jStringBase::create(get()->data_+start, ilen );
258 }
259 jString substr(std::size_t start, std::size_t len) const {
260 if( start >= get()->len_ ){
261 return get_empty();
262 }
263 auto ilen=std::min(get()->len_-start, len);
264 if(ilen == get()->len_)
265 return *this;
266 return jStringBase::create(get()->data_+start,ilen);
267 }

269 jString pad(std::size_t length, char pad_char = ' ' ) const {
270 if( length <= len())
271 return *this;
272
273 jStringBase * new_stringBase = jStringBase::create(get()->data_, length );
274
275 char * pad_ptr = new_stringBase->data_ + len();
276 for( ; pad_ptr < new_stringBase->data_ + length; ++ pad_ptr ) {
277 *pad_ptr = pad_char;
278 }
279 *pad_ptr = '\0';
280 return new_stringBase;
281 }

283 /*** Split string on delim, returning answers as an array of strings,
284 * once max_splits copies of the delimiter have been found, the
285 * remainder if any is a single string at the end of the array.
286 */
287 std::vector<jString> split(const char * delim, int max_splits=std::numeric_limits<int>::max() ) const {
288 std::vector<jString> answer;
289 const char * buffer = get()->data_;
290 const char * buffer_end = buffer+len();
291 const int step = strlen(delim);
292 auto position = strstr( buffer, delim );
293 if( position == nullptr ){
294 answer.push_back(*this);
295 return answer;
296 }
297 for( ;
298 position != nullptr && max_splits != 0;
299 --max_splits ){
300 jString chunk(buffer,position);
301 answer.push_back( chunk );
302 buffer += chunk.len() + step;
303 position = (buffer < buffer_end) ? strstr( buffer, delim ) : nullptr;
304 }
305 if( buffer < buffer_end ) {
306 answer.push_back(jString(buffer));
307 }
308 return answer;
309 }

354 /** Join elements in an iterable into a new string
355 * with the elements separated by our value.
356 * Elements must be valid for jString constructors.
357 * Based on Python's str.join() method
358 */
359 template<class T> jString join( const T & input ) {
360 size_t total_length = 0;
361 std::vector<jString> strings; // No guarantee that input can be parsed twice
362 for( jString bit : input ) {
363 strings.push_back( bit );
364 total_length += bit.len() + len();
365 }
366 total_length = total_length + 1 - len();
367 jStringBase * answer = jStringBase::create(total_length);
368 char * out = const_cast<char *>(answer->data_);
369 bool first = true;
370 for( jString bit : strings ) {
371 if( first )
372 first = false;
373 else {
374 memmove(out, data(), len());
375 out+=len();
376 }
377 memmove(out, bit.data(), bit.len());
378 out+=bit.len();
379 }
380 *out='\0';
381 return answer;
382 }
