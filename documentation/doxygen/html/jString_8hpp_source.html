40 /**
41 * Implementation of a reference counted string
42 * This string is lighweight with the string class
43 * itself being little more than a slightly smart pointer.
44 *
45 * Primarily created to satisfy a need I have for
46 * dynamic invariant strings in a re2cc/lemon based project
47 * which will be passing a ridiculous number of strings around.
48 */
49
50 // If you get strange crashes, enable this #define
51 // To double the number of heap allocations / frees
52 // but do it safely
53 //#define SAFE_JSTRING 1
54
55 struct jStringBase {
56 /**
57 * A class to store the string data.
58 */
59 private:
60 friend class jString;
61 friend class CountedPointer<jStringBase>; // = jString's parent class;
62 friend std::ostream& operator<<(std::ostream& os, const class jString & string);
63 // no copy constructor
64 jStringBase( const jStringBase & ) = delete;
65 // suppress assignment into the string
66 jStringBase& operator = (const jStringBase&) = delete;
67 // Implement the requirements for CountedPointer
68 int counter_;
69 inline void CountedPointerAttach() { ++counter_; }
70 inline void CountedPointerDetach() { if( !--counter_) delete this;}
71 // ctor private so only jString can access it
72 jStringBase( const char * data, std::size_t len )
73 : counter_(0)
74#ifdef SAFE_JSTRING
75 , m_data( new char[len+1])
76#endif
77 , m_len( len )
78 {
79 strncpy(m_data, data, len);
80 m_data[len] = '\0';
81 }
82 static jStringBase * create(const char * data, std::size_t len) {
83#ifdef SAFE_JSTRING
84 // Double create on heap
85 return new jStringBase(data, len);
86#else
87 void * new_store = new unsigned char[len + sizeof(jStringBase)];
88 jStringBase * new_object= new (new_store) jStringBase(data, len);
89 return new_object;
90#endif
91 }
92#ifdef SAFE_JSTRING
93 virtual ~jStringBase() {
94 delete m_data;
95 }
96#endif
97 std::size_t m_len;
98#ifdef SAFE_JSTRING
99 // Not the most efficient, but portable.
100 char * m_data;
101#else
102 // **** This MUST be the last data member in the struct as
103 // we deliberately overrun the declared length of the array.
104 //
105 // I don't recall seeing any guarantee that struct members
106 // are laid out in declaration order, but every compiler I
107 // know works that way.
108 // If you get strange crashes, try defining SAFE_JSTRING
109 char m_data[1];
110#endif
111 public:
112 operator const char *() const { return m_data;};
113 auto len() const { return m_len;}
114 };

116 class jString: /* private */ CountedPointer<jStringBase> {
117 /***
118 * The user interface to the strings
119 */
120 private:
121 static jString get_empty() {
122 static jString empty_base("");
123 return empty_base;
124 }
125 public:
126 const char * data() const { return get()->m_data;}
127 friend std::ostream& operator<<(std::ostream& os, const class jString & string);
128 bool isvalid() const {
129 return isset();
130 }
131 // create from an existing base
132 // shouldn't really happen, but who knows
133 // what changes will occur in the future or
134 // either lemon or re2c does something weird
135 // in its generated code?
136 jString( jStringBase *base )
137 : CountedPointer<jStringBase>(base)
138 {}
139 // Create empty string
140 jString( ):
141 CountedPointer<jStringBase>(get_empty().get())
142 {}
143 // Create from c string
144 jString( const char * source ):
145 CountedPointer<jStringBase>(jStringBase::create(source, strlen(source)))
146 {}
147 // Create from start of c string
148 jString( const char * source, std::size_t len ):
149 CountedPointer<jStringBase>(jStringBase::create(source, len))
150 {}
151 // Create from two pointers to the same buffer
152 // NB: This is what re2c gives on a match
153 jString( const char * start, const char * end ):
154 CountedPointer<jStringBase>(jStringBase::create(start, end-start))
155 {}
156 // string concatenation
157 jString operator + (const jString &rhs) {
158 jStringBase * lhs_base = get();
159 if( lhs_base->m_len == 0 )
160 return rhs;
161 if( rhs.len() == 0 )
162 return *this;
163 auto len = lhs_base->m_len + rhs->m_len;
164 jStringBase * new_base = jStringBase::create(lhs_base->m_data,len );
165 strncpy(new_base->m_data + lhs_base->m_len,rhs->m_data, rhs->m_len );
166 new_base->m_data[len]='\0';
167 return new_base;
168 }
169 // extract substrings (left, right, any) with
170 // silent bounds checks.
171 // E.g. if z=="123", z.substr(2,999) will quietly return "3"
172 // NB: Const but when the whole string is to be
173 // returned, we return a copy of ourself not a
174 // whole new string. This saves creating a new
175 // jStringBase.
176 jString left(std::size_t len) const {
177 auto ilen=std::min(len,get()->m_len);
178 if(ilen == get()->m_len)
179 return *this;
180 return jStringBase::create(get()->m_data, ilen );
181 }
182 jString right(std::size_t len) const {
183 auto ilen=std::min(len,get()->m_len);
184 if(ilen == get()->m_len)
185 return *this;
186 auto start=get()->m_len-ilen;
187 return jStringBase::create(get()->m_data+start, ilen );
188 }
189 jString substr(std::size_t start, std::size_t len) const {
190 if( start >= get()->m_len ){
191 return get_empty();
192 }
193 auto ilen=std::min(get()->m_len-start, len);
194 if(ilen == get()->m_len)
195 return *this;
196 return jStringBase::create(get()->m_data+start,ilen);
197 }
198
199 jString pad(std::size_t length, char pad_char = ' ' ) const {
200 if( length <= len())
201 return *this;
202
203 jStringBase * new_stringBase = jStringBase::create(get()->m_data, length );
204
205 char * pad_ptr = new_stringBase->m_data + len();
206 for( ; pad_ptr < new_stringBase->m_data + length; ++ pad_ptr ) {
207 *pad_ptr = pad_char;
208 }
209 *pad_ptr = '\0';
210 return new_stringBase;
211 }

213 std::vector<jString> split(const char * delim, int max_splits=std::numeric_limits<int>::max() ) const {
214 std::vector<jString> answer;
215 const char * buffer = get()->m_data;
216 const char * buffer_end = buffer+len();
217 const int step = strlen(delim);
218 auto position = strstr( buffer, delim );
219 if( position == nullptr ){
220 answer.push_back(*this);
221 return answer;
222 }
223 for( ;
224 position != nullptr && max_splits != 0;
225 --max_splits ){
226 jString chunk(buffer,position);
227 answer.push_back( chunk );
228 buffer += chunk.len() + step;
229 position = (buffer < buffer_end) ? strstr( buffer, delim ) : nullptr;
230 }
231 if( buffer < buffer_end ) {
232 answer.push_back(jString(buffer));
233 }
234 return answer;
235 }
236 bool operator < (const jString &rhs) const {
237 return strcmp(get()->m_data, rhs->m_data) < 0;
238 }
239 bool operator <= (const jString &rhs) const {
240 return strcmp(get()->m_data, rhs->m_data) <= 0;
241 }
242 bool operator > (const jString &rhs) const {
243 return strcmp(get()->m_data, rhs->m_data) > 0;
244 }
245 bool operator >= (const jString &rhs) const {
246 return strcmp(get()->m_data, rhs->m_data) >= 0;
247 }
248 bool operator == (const jString &rhs) const {
249 return strcmp(get()->m_data, rhs->m_data) == 0;
250 }
251 bool operator != (const jString &rhs) const {
252 return strcmp(get()->m_data, rhs->m_data) != 0;
253 }
254 size_t len() const { return get()->len();}
255
256 operator const char *() const { return (*this)->m_data;};
257 };
