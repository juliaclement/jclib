38 /**
39 * Implementation of a reference counted string
40 * This string is lighweight with the string class
41 * itself being little more than a slightly smart pointer.
42 *
43 * Primarily created to satisfy a need I have for
44 * dynamic invariant strings in a re2cc/lemon based project
45 * which will be passing a ridiculous number of strings around.
46 */
47
48 class jStringBase: public CountedPointerTarget {
49 /**
50 * A class to store the string data.
51 */
52 private:
53 friend class jString;
54 friend std::ostream& operator<<(std::ostream& os, const class jString & string);
55 std::size_t m_len;
56 // Not the most efficient, but portable.
57 char * m_data;
58 // suppress assignment into the string
59 jStringBase & operator=(const jStringBase &);
60 // ctor private so only jString can access it
61 jStringBase( const char * data, std::size_t len ){
62 m_data = new char[len+1];
63 m_len = len;
64 strncpy(m_data, data, len);
65 m_data[len] = '\0';
66 }
67 public:
68 operator const char *() const { return m_data;};
69 auto len() const { return m_len;}
70 };
71
72 class jString: /* private */ CountedPointer<jStringBase> {
73 /***
74 * The user interface to the strings
75 */
76 private:
77 static jString get_empty() {
78 static jString empty_base("");
79 return empty_base;
80 }
81 const char * data() const { return get()->m_data;}
82 friend std::ostream& operator<<(std::ostream& os, const class jString & string);
83 public:
84 // create from an existing base
85 // shouldn't really happen, but who knows
86 // what changes will occur in the future or
87 // either lemon or re2c does something weird
88 // in its generated code?
89 jString( jStringBase *base )
90 : CountedPointer<jStringBase>(base)
91 {}
92 // Create from c string
93 jString( const char * source ):
94 CountedPointer<jStringBase>(new jStringBase(source, strlen(source)))
95 {}
96 // Create from start of c string
97 jString( const char * source, std::size_t len ):
98 CountedPointer<jStringBase>(new jStringBase(source, len))
99 {}
100 // Create from two pointers to the same buffer
101 // NB: This is what re2c gives on a match
102 jString( const char * start, const char * end ):
103 CountedPointer<jStringBase>(new jStringBase(start, end-start))
104 {}
105 // string concatenation
106 jString operator + (const jString &rhs) {
107 jStringBase * lhs_base = get();
108 if( lhs_base->m_len == 0 )
109 return *this;
110 auto len = lhs_base->m_len + rhs->m_len;
111 jStringBase * new_base = new jStringBase(lhs_base->m_data,len + 1 );
112 strncpy(new_base->m_data + lhs_base->m_len,rhs->m_data, rhs->m_len );
113 new_base->m_data[len]='\0';
114 return new_base;
115 }
116 // extract substrings (left, right, any) with
117 // silent bounds checks.
118 // E.g. if z=="123", z.substr(2,999) will quietly return "3"
119 // NB: Const but when the whole string is to be
120 // returned, we return a copy of ourself not a
121 // whole new string. This saves creating a new
122 // jStringBase.
123 jString left(std::size_t len) const {
124 auto ilen=std::min(len,get()->m_len);
125 if(ilen == get()->m_len)
126 return *this;
127 return new jStringBase(get()->m_data, ilen );
128 }
129 jString right(std::size_t len) const {
130 auto ilen=std::min(len,get()->m_len);
131 if(ilen == get()->m_len)
132 return *this;
133 auto start=get()->m_len-ilen;
134 return new jStringBase(get()->m_data+start, ilen );
135 }
136 jString substr(std::size_t start, std::size_t len) const {
137 if( start >= get()->m_len ){
138 return get_empty();
139 }
140 auto ilen=std::min(get()->m_len-start, len);
141 if(ilen == get()->m_len)
142 return *this;
143 return new jStringBase(get()->m_data+start,ilen);
144 }
145 bool operator < (const jString &rhs) const {
146 return strcmp(get()->m_data, rhs->m_data) < 0;
147 }
148 bool operator > (const jString &rhs) const {
149 return strcmp(get()->m_data, rhs->m_data) > 0;
150 }
151 bool operator == (const jString &rhs) const {
152 return strcmp(get()->m_data, rhs->m_data) == 0;
153 }
154 operator const char *() const { return (*this)->m_data;};
155 };
156

158 std::ostream& operator<<(std::ostream& os, const jString & string) {
159 os << string.data();
160 return os;
161 }
162};
