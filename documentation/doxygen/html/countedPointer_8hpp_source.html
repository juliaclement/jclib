34/**
35 * This is an implementation of an intrusive counted pointed.
36 * The typename T must expose two public methods CountedPointerAtttach and
37 * CountedPointerDetach. The expected behaviour is that these will maintain
38 * a count of attached pointers and self delete when the counter reaches zero
39**/
40template< typename T> class CountedPointer {
41private:
42 T * ptr_;
43 inline void postAssign() {
44 if( ptr_ != nullptr )
45 ptr_->CountedPointerAttach();
46 }
47public:
48 /**
49 * Test if the pointer is set to nullptr
50 * @return true if the internal pointer is nullptr
51 **/
52 bool isnull() const { return ptr_ == nullptr; }
53 /**
54 * Test if the pointer is set to an object
55 * @return true if the internal pointer is not nullptr
56 **/
57 bool isset() const { return ptr_ != nullptr; }
58
59 /**
60 * @return the address of the object pointed at
61 */
62 T * get() const { return const_cast< T *>(ptr_); }
63 /**
64 * retrieve a reference to the address of the object pointed at
65 */
66 T & operator *() const { return const_cast< T &>(*ptr_); }
67 T * operator ->() const { return const_cast< T *>(ptr_); }
68 /**
69 * Set the internal value. Attaching / Detaching as requried
70 * occasionaally useful as a replacement for the assignment operator.
71 * @param ptr the new value to set
72 */
73 inline void set( T * const ptr) {
74 // Note: We Attach before Detach so A=A (with attach count==1)
75 // doesn't cause the attach count to temporarily zero and have
76 // the target delete itself;
77 T * tmp_ptr=ptr_;
78 ptr_ = const_cast< T*>(ptr);
79 postAssign();
80 if( tmp_ptr != nullptr ) {
81 tmp_ptr->CountedPointerDetach();
82 }
83 }
84 /**
85 * Set the internal value. Attaching / Detaching as requried
86 * occasionaally useful as a replacement for the assignment operator.
87 * @param ptr the new value to set
88 */
89 inline void set( const CountedPointer<T> & old ) {
90 set( old.ptr_);
91 }
92 /**
93 * Set the internal value. Attaching / Detaching as requried
94 * @param ptr the new value to set
95 * @return ourself
96 */
97 class CountedPointer<T> & operator =( T * const ptr ) {
98 set( ptr );
99 return *this;
100 }
101 /**
102 * Set the internal value. Attaching / Detaching as requried
103 * @param ptr the new value to set
104 * @return ourself
105 */
106 class CountedPointer<T> & operator =( const CountedPointer<T> & ptr ) {
107 set( ptr.ptr_ );
108 return *this;
109 }
110 CountedPointer( ) : ptr_( nullptr) { postAssign(); }
111 CountedPointer( T * ptr ) : ptr_( ptr) { postAssign(); }
112 CountedPointer( const class CountedPointer<T>& old ) : ptr_( old.ptr_) { postAssign(); }
113 ~CountedPointer() { if( ptr_ != nullptr ) ptr_->CountedPointerDetach(); ptr_=nullptr; }
114};

116/**
117 * A simple class that satisfies requirements for targets of CountedPointer
118 *
119 */
120
121class CountedPointerTarget {
122 private:
123 int counter_;
124 public:
125 inline void CountedPointerAttach() { ++counter_; }
126 inline void CountedPointerDetach() { if( !--counter_) delete this;}
127 inline int CountedPointersAttached() const { return counter_; }
128 CountedPointerTarget() : counter_(0) {}
129 CountedPointerTarget(const CountedPointerTarget &) : counter_(0) {}
130 // virtual destructor to ensure derrived classes are destroyed properly
131 virtual ~CountedPointerTarget() {}
132};

134/**
135 * A template child class that permits automatic & static variables to have
136 * counted pointers.
137 *
138 * The user must ensure that all CountedPointers to instances of this class
139 * are deleted before the target.
140 *
141 * NB: It is not generally safe to have CountedPointers point at automatic or
142 * static variables as when the pointer count reaches zero an attempt will
143 * be made to delete the variable.
144 */
145template<typename T> class CountedPointerSafe : public T {
146 public:
147 template< typename ... Types > CountedPointerSafe( const Types&...args )
148 : T(args...)
149 { T::CountedPointerAttach(); }
150};
151
152};
